cmake_minimum_required(VERSION 3.22)

project(PixelEngineNative
  VERSION 0.1.0
  DESCRIPTION "PixelEngine native dependency superbuild"
  LANGUAGES C CXX
)

include(ExternalProject)

if(NOT DEFINED PIXELENGINE_TARGET_RID OR PIXELENGINE_TARGET_RID STREQUAL "")
  message(FATAL_ERROR "PIXELENGINE_TARGET_RID must be set, for example win-x64.")
endif()

set(PIXELENGINE_NATIVE_OUT_DIR "${CMAKE_CURRENT_SOURCE_DIR}/out/${PIXELENGINE_TARGET_RID}")
set(PIXELENGINE_BOX2D_SOURCE_DIR "${CMAKE_CURRENT_SOURCE_DIR}/box2d")
set(PIXELENGINE_BOX2D_COMMON_ARGS
  -DCMAKE_BUILD_TYPE=${CMAKE_BUILD_TYPE}
  -DBOX2D_SAMPLES=OFF
  -DBOX2D_UNIT_TESTS=OFF
  -DBOX2D_BENCHMARKS=OFF
  -DBOX2D_DOCS=OFF
  -DBOX2D_PROFILE=OFF
  -DBOX2D_VALIDATE=OFF
  -DBOX2D_DISABLE_SIMD=OFF
)

set(PIXELENGINE_BOX2D_TARGET_ARGS)
if(CMAKE_TOOLCHAIN_FILE)
  list(APPEND PIXELENGINE_BOX2D_TARGET_ARGS
    -DCMAKE_TOOLCHAIN_FILE:FILEPATH=${CMAKE_TOOLCHAIN_FILE}
  )
endif()

if(CMAKE_SYSTEM_NAME)
  list(APPEND PIXELENGINE_BOX2D_TARGET_ARGS
    -DCMAKE_SYSTEM_NAME=${CMAKE_SYSTEM_NAME}
  )
endif()

if(CMAKE_SYSTEM_PROCESSOR)
  list(APPEND PIXELENGINE_BOX2D_TARGET_ARGS
    -DCMAKE_SYSTEM_PROCESSOR=${CMAKE_SYSTEM_PROCESSOR}
  )
endif()

if(CMAKE_OSX_ARCHITECTURES)
  list(APPEND PIXELENGINE_BOX2D_TARGET_ARGS
    -DCMAKE_OSX_ARCHITECTURES=${CMAKE_OSX_ARCHITECTURES}
  )
endif()

ExternalProject_Add(box2d_shared
  SOURCE_DIR "${PIXELENGINE_BOX2D_SOURCE_DIR}"
  BINARY_DIR "${CMAKE_BINARY_DIR}/box2d-shared"
  INSTALL_COMMAND ""
  CMAKE_ARGS
    ${PIXELENGINE_BOX2D_COMMON_ARGS}
    ${PIXELENGINE_BOX2D_TARGET_ARGS}
    -DBUILD_SHARED_LIBS=ON
    -DCMAKE_MSVC_RUNTIME_LIBRARY=MultiThreadedDLL
    -DCMAKE_RUNTIME_OUTPUT_DIRECTORY=${PIXELENGINE_NATIVE_OUT_DIR}/shared
    -DCMAKE_LIBRARY_OUTPUT_DIRECTORY=${PIXELENGINE_NATIVE_OUT_DIR}/shared
    -DCMAKE_ARCHIVE_OUTPUT_DIRECTORY=${PIXELENGINE_NATIVE_OUT_DIR}/shared
)

ExternalProject_Add(box2d_static
  SOURCE_DIR "${PIXELENGINE_BOX2D_SOURCE_DIR}"
  BINARY_DIR "${CMAKE_BINARY_DIR}/box2d-static"
  INSTALL_COMMAND ""
  CMAKE_ARGS
    ${PIXELENGINE_BOX2D_COMMON_ARGS}
    ${PIXELENGINE_BOX2D_TARGET_ARGS}
    -DBUILD_SHARED_LIBS=OFF
    -DCMAKE_MSVC_RUNTIME_LIBRARY=MultiThreaded
    -DCMAKE_RUNTIME_OUTPUT_DIRECTORY=${PIXELENGINE_NATIVE_OUT_DIR}/static
    -DCMAKE_LIBRARY_OUTPUT_DIRECTORY=${PIXELENGINE_NATIVE_OUT_DIR}/static
    -DCMAKE_ARCHIVE_OUTPUT_DIRECTORY=${PIXELENGINE_NATIVE_OUT_DIR}/static
)
